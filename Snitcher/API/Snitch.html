def initialize(payload)
  @token     = payload["token"]
  @name      = payload["name"]
  @tags      = payload["tags"]
  @status    = payload["status"]
  @interval  = payload["type"]["interval"]
  @notes     = payload["notes"]

  @created_at    = payload["created_at"]
  @check_in_url  = payload["check_in_url"]
  @checked_in_at = payload["checked_in_at"]
end
